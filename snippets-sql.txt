    varchar_column varchar(10),
    char_column char(10),
    text_column text
);

SELECT *
FROM information_schema.columns
WHERE table_schema = 'your_schema'
AND table_name   = 'your_table'
;

# varchar is self-limiting up until the final character, no padding added
